create table hotel(hotel_id int primary key,hotel_name varchar(255),rating int ,description varchar(255));


create table room_type_table(room_type_id int primary key,hotel_id int references hotel(hotel_id),type_name varchar(100), base_price int , capacity int );

create table booking_table(booking_id int generated by default as identity (start with 100), 
user_id int references user_table(user_id), room_type_id int references room_type_table(room_type_id),checkin_date timestamp, checkput_date timestamp , total_nights int, total_price dec(10,2), primary key(booking_id));

create table food_table(food_id varchar(100), food_name varchar(255), price int ,category varchar(100), isavailable bool );

create table booking_FoodOrder_table(booking_id int references booking_table(booking_id) , food_id varchar(100) references food_table(food_id), quantity int, primary key(booking_id, food_id));

create table facility_table(facility_id int generated always as identity primary key ,facility_name varchar(100));
create table hotel_facility(hotel_id int references hotel(hotel_id), facility_id int references facility_table(facility_id), primary key(hotel_id, facility_id))
